import pandas as pd
from bs4 import BeautifulSoup
import requests

url = "https://cf-courses-data.s3.us.cloud-object-storage.appdomain.cloud/IBMDeveloperSkillsNetwork-PY0220EN-SkillsNetwork/labs/project/revenue.htm"

response = requests.get(url)

if response.status_code == 200:
    soup = BeautifulSoup(response.text, 'html.parser')

    table = soup.find_all('table')[1]  # Use [1] to select the second table on the page

    dates = []
    revenues = []

    for row in table.find_all('tr')[1:]: 
        cols = row.find_all('td')
        if len(cols) == 2:
            date = cols[0].text.strip()
            revenue = cols[1].text.strip().replace(',', '').replace('$', '')  # Clean the data
            dates.append(date)
            revenues.append(revenue)

    tesla_revenue = pd.DataFrame({'Date': dates, 'Revenue': revenues})

    print(tesla_revenue.tail(5))
else:
    print("Failed to retrieve the webpage. Status code:", response.status_code)

import yfinance as yf
import matplotlib.pyplot as plt

# Define the make_graph function
def make_graph(stock_data, stock_name, title):
    plt.figure(figsize=(12, 6))

    plt.plot(stock_data['Date'], stock_data['Close'], label=f'{stock_name} Stock Price', color='b')

    plt.title(title)
    plt.xlabel('Date')
    plt.ylabel('Stock Price')
    plt.xticks(rotation=45)

    plt.legend()

    plt.grid(True)
    plt.show()
